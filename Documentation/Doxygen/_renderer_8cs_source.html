69  public void BuildDisplay(DrawingContext ctx, int lvl, int arg2, Dictionary<string, string> dict, Dictionary<string, int> scores)
70  {
71  this.level = lvl;
72  if (ctx != null)
73  {

78  case 1000: this.WinGame(ctx); break;
79  case 30: this.HighScore(ctx, scores); break;
80  case -1: this.EndGame(ctx); break;
81  case 100: this.GetName(ctx, dict); break;
82  case 101: this.FirstScene(ctx); break;
83  case 10:
84  if (dict != null)
85  {
86  this.LoadScreen(ctx, dict, arg2);
87  }
88  else
89  {
90  this.LoadScreen(ctx, null, arg2);
91  }
92 
93  break;
94  }
