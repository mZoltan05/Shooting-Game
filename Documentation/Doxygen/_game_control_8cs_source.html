271  private void HighScore()
272  {
273  Dictionary<string, int> tempList = this.repository.GetAllScores();
274  this.scores = new Dictionary<string, int>();
275  KeyValuePair<string, int> maxitem;
276  while (tempList.Count > 0)
277  {
278  maxitem = new KeyValuePair<string, int>(" ", 0);
279  foreach (KeyValuePair<string, int> item in tempList)
280  {
281  if (Convert.ToInt32(item.Value) > Convert.ToInt32(maxitem.Value))
282  {
283  maxitem = item;
284  }
285  }
286 
287  this.scores.Add(maxitem.Key, maxitem.Value);
288  tempList.Remove(maxitem.Key);
289  }
290  }
